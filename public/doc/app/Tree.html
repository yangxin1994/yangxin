def initialize(*args)
        args_size = args.length
        @root, @nodes, @edges, @constrains = nil, {}, {}, {}
        if args_size == 0
                # the tree is initialized as an empty tree, do not need to do anything
        elsif args_size == 2
                # the tree is initialized by a node in another tree
                raise ArgumentError, 'the first parameter should be a Tree type instance' if args[0].class != "Tree"
                tree, node_id = *args
                @root = tree.nodes[node_id]
                raise ArgumentError, "can not find the node '#{node_id} in the Tree #{tree.to_s}'" if @root == nil
                init_by_node(tree, node_id)
                @edges[node_id][:parent] = -1
        else
                raise ArgumentError, 'the initialize method of Tree accepts no parameter or two parameter'
        end
end

def add_attr(node_id, attr)
        return ErrorEnum::NODE_NOT_EXIST if @nodes[node_id].nil?
        attr.each_key do |key|
                return ErrorEnum::ATTR_NOT_EXIST if not(Object::const_get(@nodes[node_id][:node_type]).has_attr?(key))
        end
        @nodes[node_id][:attr] = @nodes[node_id][:attr].merge(attr)
end

def deserialize(object)
        @nodes = object[:nodes]
        @edges = object[:edges]
        @root = object[:root]
        @constrains = Hash.new
        object[:constrains].each do |node_id, constrain_array|
                constrains[node_id] = Array.new
                constrain_array.each do |constrain|
                        @constrains[node_id] << Constrain.deserialize(constrain)
                end
        end
end

def insert_exist_child_node(parent_id, child_index, tree, node_id)
        return ErrorEnum::INDEX_OVERFLOW if !(-(@edges[parent_id][:children].length + 1)..@edges[parent_id][:children].length).include?(child_index)
        old_node_id = node_id
        node_id = SecureRandom.uuid
        @nodes[node_id] = tree.nodes[old_node_id]
        @nodes[node_id][:node_id] = node_id
        @edges[parent_id][:children].insert(child_index, node_id)
        @edges[node_id][:parent] = parent_id
        @edges[node_id][:children] = tree.edges[old_node_id][:children].clone
        @constrains[node_id] = tree.constrains[old_node_id].clone
end
