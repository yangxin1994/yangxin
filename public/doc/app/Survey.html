def self.save_meta_data(email, survey)
        if survey["new_survey"] == true
                # this is a new survey, should save it
                survey["owner_email"] = email
                survey.save
                return 1
        else
                # this is an existing survey, should update its meta data
                survey = Survey.find_by_id(survey["_id"])
                return ErrorEnum::NotExist if survey == nil
                return ErrorEnum::Unauthorized if survey.owner_email != owner_email
                survey.update_attributes(:caption => survey["caption"], :sub_caption => survey["sub_caption"], :description => survey["description"], :keywords => survey["keywords"], :header => survey["header"], :footer => survey["footer"], :thanks_message => survey["thanks_message"], :style_sheet => survey["style_sheet"] )
                return 1
        end
end
