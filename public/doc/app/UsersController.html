def reset_password
        reset_password_retval = @current_user.reset_password(params["old_password"], params["new_password"])
        case reset_password_retval
        when ErrorEnum::WrongPassword
                flash[:notice] = "密码错误"
                redirect_to reset_password_path and return
        else
                flash[:notice] = "密码已重置"
                redirect_to home_path and return
        end
end

def update_information
        @current_user.update_information(params[:user_information])
        respond_to do |format|
                format.json  { render :json => 1 and return }
        end
end
